## 1 ── Measles cases -------------------------------------------------------
measles_df <-
  read_csv("measles_cases.csv", show_col_types = FALSE) |>
  clean_names() |>
  select(state, confirmed_cases) |>
  mutate(state = str_trim(state))

## 2 ── Population ----------------------------------------------------------
pop_raw <-
  read_excel("SCPRC-EST2024-18+POP.xlsx",
             skip      = 4,
             col_names = FALSE)

population_df <-
  tibble(
    state      = str_trim(str_remove(pop_raw[[1]], "^\\.+")),
    population = as.numeric(pop_raw[[2]])
  ) |>
  filter(!state %in% c("United States", "Northeast", "Midwest",
                       "South", "West", "Puerto Rico")) |>
  drop_na(population)

## 3 ── Merge & derive severity --------------------------------------------
merged_df <-
  population_df |>
  left_join(measles_df, by = "state") |>
  mutate(
    confirmed_cases = replace_na(confirmed_cases, 0),
    severity        = round((confirmed_cases / population) * 1e5, 2),
    severity_bucket = case_when(
      severity == 0 ~ NA_character_,
      severity <= 1 ~ "1",
      severity <= 2 ~ "2",
      severity <= 3 ~ "3",
      severity <= 4 ~ "4",
      TRUE          ~ "5"
    )
  )

## 4 ── Five-colour palette -------------------------------------------------
pal <- c("1" = "#fef0d9",
         "2" = "#fdcc8a",
         "3" = "#fc8d59",
         "4" = "#e34a33",
         "5" = "#b30000")

## 5 ── Plot with black borders & boxed legend --------------------------------
ggplot(map_merged) +
  geom_sf(aes(fill = severity_bucket),          # polygons
          colour = "black", size = 0.4) +       # ← black state borders
  scale_fill_manual(
    values   = pal,
    na.value = "white",
    drop     = FALSE,
    name     = "Measles cases\n(per 100 K)"
  ) +
  geom_text(                                    # state abbreviations
    data        = label_df,
    aes(x = x, y = y, label = abbr),
    inherit.aes = FALSE,
    fontface    = "bold",
    size        = 3
  ) +
  coord_sf() +
  theme_void() +
  theme(
    legend.position   = "right",
    legend.background = element_rect(           # ← legend box border
      colour = "black", fill = "white", linewidth = 0.4
    )
  ) +
  labs(title = "Measles Severity by State (per 100 K) – 21 June 2025")
